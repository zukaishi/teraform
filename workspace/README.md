# 作業用環境を構築する
terraform apply<br>
<br>

# ping
ping [public_ip]<br>
<br>

# ssh接続(Amazon Linux)
ssh -i ./workspace ec2-user@[ip_address]]<br>
# ssh接続(ubuntu)
ssh -i ./workspace ubuntu@[ip_address]]<br>
<br>

# 作業用環境を撤去する
terraform destroy<br>
<br>
